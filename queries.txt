/* ===================== TEMPLATE FUNCTION (COPY PASTE LOGIC) ===================== */

-- ---------- PATIENTS ----------
CREATE TABLE patients_5k (
    patient_id INT PRIMARY KEY,
    first_name VARCHAR(50),
    last_name VARCHAR(50),
    date_of_birth DATE,
    gender VARCHAR(10),
    phone VARCHAR(15),
    email VARCHAR(100),
    address TEXT,
    blood_group VARCHAR(5),
    emergency_contact VARCHAR(15),
    created_at TIMESTAMP
);

CREATE TABLE patients_10k (LIKE patients_5k INCLUDING ALL);
CREATE TABLE patients_25k (LIKE patients_5k INCLUDING ALL);


-- ---------- DOCTORS (FIXED) ----------
CREATE TABLE doctors_5k (
    doctor_id INT PRIMARY KEY,
    first_name VARCHAR(50),
    last_name VARCHAR(50),
    phone VARCHAR(15),
    email VARCHAR(100),
    years_of_experience INT,
    license_number VARCHAR(50),
    department_id INT,
    created_at TIMESTAMP,

    FOREIGN KEY (department_id) REFERENCES departments(department_id)
);

CREATE TABLE doctors_10k (LIKE doctors_5k INCLUDING ALL);
CREATE TABLE doctors_25k (LIKE doctors_5k INCLUDING ALL);


-- ---------- DOCTOR SPECIALIZATIONS ----------
CREATE TABLE doctor_specializations_5k (
    id INT PRIMARY KEY,
    doctor_id INT,
    specialization VARCHAR(100),

    FOREIGN KEY (doctor_id) REFERENCES doctors_5k(doctor_id)
);

CREATE TABLE doctor_specializations_10k (
    LIKE doctor_specializations_5k INCLUDING ALL,
    FOREIGN KEY (doctor_id) REFERENCES doctors_10k(doctor_id)
);

CREATE TABLE doctor_specializations_25k (
    LIKE doctor_specializations_5k INCLUDING ALL,
    FOREIGN KEY (doctor_id) REFERENCES doctors_25k(doctor_id)
);


-- ---------- APPOINTMENTS ----------
CREATE TABLE appointments_5k (
    appointment_id INT PRIMARY KEY,
    patient_id INT,
    doctor_id INT,
    appointment_date DATE,
    appointment_time TIME,
    status VARCHAR(20),
    reason_for_visit TEXT,
    created_at TIMESTAMP,

    FOREIGN KEY (patient_id) REFERENCES patients_5k(patient_id),
    FOREIGN KEY (doctor_id) REFERENCES doctors_5k(doctor_id)
);

CREATE TABLE appointments_10k (
    LIKE appointments_5k INCLUDING ALL,
    FOREIGN KEY (patient_id) REFERENCES patients_10k(patient_id),
    FOREIGN KEY (doctor_id) REFERENCES doctors_10k(doctor_id)
);

CREATE TABLE appointments_25k (
    LIKE appointments_5k INCLUDING ALL,
    FOREIGN KEY (patient_id) REFERENCES patients_25k(patient_id),
    FOREIGN KEY (doctor_id) REFERENCES doctors_25k(doctor_id)
);


-- ---------- PRESCRIPTIONS ----------
CREATE TABLE prescriptions_5k (
    prescription_id INT PRIMARY KEY,
    appointment_id INT,
    medication_name VARCHAR(100),
    dosage VARCHAR(50),
    frequency VARCHAR(50),
    duration_days INT,
    notes TEXT,
    issued_date DATE,

    FOREIGN KEY (appointment_id) REFERENCES appointments_5k(appointment_id)
);

CREATE TABLE prescriptions_10k (
    LIKE prescriptions_5k INCLUDING ALL,
    FOREIGN KEY (appointment_id) REFERENCES appointments_10k(appointment_id)
);

CREATE TABLE prescriptions_25k (
    LIKE prescriptions_5k INCLUDING ALL,
    FOREIGN KEY (appointment_id) REFERENCES appointments_25k(appointment_id)
);


-- ---------- BILLING ----------
CREATE TABLE billing_5k (
    bill_id INT PRIMARY KEY,
    appointment_id INT,
    consultation_fee DECIMAL(10,2),
    medicine_charges DECIMAL(10,2),
    lab_charges DECIMAL(10,2),
    total_amount DECIMAL(10,2),
    payment_status VARCHAR(20),
    payment_method VARCHAR(20),
    bill_date TIMESTAMP,

    FOREIGN KEY (appointment_id) REFERENCES appointments_5k(appointment_id)
);

CREATE TABLE billing_10k (
    LIKE billing_5k INCLUDING ALL,
    FOREIGN KEY (appointment_id) REFERENCES appointments_10k(appointment_id)
);

CREATE TABLE billing_25k (
    LIKE billing_5k INCLUDING ALL,
    FOREIGN KEY (appointment_id) REFERENCES appointments_25k(appointment_id)
);




-- =========================

CREATE INDEX idx_patients_email_5k ON patients_5k(email);

-- =========================
CREATE INDEX idx_doc_spec_doctor_id_5k ON doctor_specializations_5k(doctor_id);
CREATE INDEX idx_doc_spec_specialization_5k ON doctor_specializations_5k(specialization);

-- =========================
-- APPOINTMENTS (MOST CRITICAL)
-- =========================
CREATE INDEX idx_appointments_patient_id_5k ON appointments_5k(patient_id);

CREATE INDEX idx_appointments_status_5k ON appointments_5k(status);

-- composite
CREATE INDEX idx_appointments_patient_date_5k 
ON appointments_5k(patient_id, appointment_date);

CREATE INDEX idx_appointments_doctor_date_5k 
ON appointments_5k(doctor_id, appointment_date);

-- =========================
-- BILLING (REVENUE BOTTLENECK)
-- =========================
CREATE INDEX idx_billing_appointment_id_5k ON billing_5k(appointment_id);
CREATE INDEX idx_billing_date_5k ON billing_5k(bill_date);
CREATE INDEX idx_billing_payment_status_5k ON billing_5k(payment_status);

-- composite
CREATE INDEX idx_billing_date_status_5k 
ON billing_5k(bill_date, payment_status);

-- =========================
-- PRESCRIPTIONS
-- =========================
CREATE INDEX idx_prescriptions_appointment_id_5k ON prescriptions_5k(appointment_id);

-- DEPARTMENTS (NO SUFFIX)
-- =========================

CREATE SEQUENCE IF NOT EXISTS patients_5k_id_seq;

SELECT setval('patients_5k_id_seq', (SELECT MAX(patient_id) FROM patients_5k));

ALTER TABLE patients_5k 
ALTER COLUMN patient_id SET DEFAULT nextval('patients_5k_id_seq');



CREATE SEQUENCE IF NOT EXISTS doctors_5k_id_seq;

SELECT setval('doctors_5k_id_seq', (SELECT MAX(doctor_id) FROM doctors_5k));

ALTER TABLE doctors_5k 
ALTER COLUMN doctor_id SET DEFAULT nextval('doctors_5k_id_seq');




CREATE SEQUENCE IF NOT EXISTS prescription_5k_id_seq;

SELECT setval('prescription_5k_id_seq', (SELECT MAX(prescription_id) FROM prescriptions_5k));

ALTER TABLE prescriptions_5k 
ALTER COLUMN prescription_id SET DEFAULT nextval('prescription_5k_id_seq');




CREATE SEQUENCE IF NOT EXISTS doctor_specializations_5k_id_seq;

SELECT setval('doctor_specializations_5k_id_seq', (SELECT MAX(id) FROM doctor_specializations_5k));

ALTER TABLE doctor_specializations_5k 
ALTER COLUMN id SET DEFAULT nextval('doctor_specializations_5k_id_seq');


CREATE SEQUENCE IF NOT EXISTS appointments_5k_id_seq;

SELECT setval('appointments_5k_id_seq', COALESCE((SELECT MAX(appointment_id) FROM appointments_5k), 1));

ALTER TABLE appointments_5k 
ALTER COLUMN appointment_id SET DEFAULT nextval('appointments_5k_id_seq');



CREATE SEQUENCE IF NOT EXISTS billing_5k_id_seq;

SELECT setval('billing_5k_id_seq', COALESCE((SELECT MAX(bill_id) FROM billing_5k), 1));

ALTER TABLE billing_5k 
ALTER COLUMN bill_id SET DEFAULT nextval('billing_5k_id_seq');
